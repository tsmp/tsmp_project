
     private void button1_Click(object sender, EventArgs e)
        {
            Save(textBox1.Text);
        }
 
        private void button2_Click(object sender, EventArgs e)
        {
            textBox1.Text = LoadData();
        }
 
        private static void Save(string data)
        {
            StreamWriter writer = new StreamWriter("data.txt", true, Encoding.GetEncoding("windows-1251"));
            writer.Write(data);
            writer.Close();
        }
 
        private static string LoadData()
        {
            StreamReader reader = new StreamReader("data.txt", Encoding.GetEncoding("windows-1251"));
            return reader.ReadToEnd();
        }
